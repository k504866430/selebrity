Page Object Pattern

import static com.selebrity.seleniumsupport.Browser.*;
@Page(url="https://www.google.com", verifyTitle="Google")
public class GoogleSearchPage extends Page{
  @FindBy(name="q")
  private WebElement q;

  public void search(String s){
    $(q).value(s).submit();
  }
}

import static com.selebrity.seleniumsupport.Browser.*;
@Page
@VerifyAt(titleContains="Google Search")
public class GoogleResultsPage extends Page{
  public void assertResult(String link){
    $("#search li.g a[data-href='"+link+"']").shouldNotBeNull();
  }
}

{
  GoogleSearchPage searchPage = Browser.go(GoogleSearchPage.class);
  searchPage.search("selebrity");
  GoogleResultsPage resultsPage = Browser.verifyAt(GoogleResultsPage.class);
  resultsPage.assertResult("https://github.com/k504866430");
}
